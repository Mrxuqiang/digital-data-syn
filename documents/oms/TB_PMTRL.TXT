表结构目的:
================================================================================

drop sequence SEQ_TB_PMTRL;
drop index AK_TB_PMTRL;
drop table TB_PMTRL;
create table TB_PMTRL  (
   PMTRL_ID             INTEGER                         not null,  /*交款返现生效时间ID*/
   PMTRL001             INTEGER                         not null,  /*交款返现促销单ID  */
   PMTRL002             DATE                            not null,  /*开始日期          */
   PMTRL003             VARCHAR2(8)                     not null,  /*开始时间          */
   PMTRL004             DATE                            not null,  /*结束日期          */
   PMTRL005             VARCHAR2(8)                     not null,  /*结束时间          */
   CREATE_USER          VARCHAR2(12),                              /*建立人员          */
   USER_GROUP           VARCHAR2(12),                              /*建立人员部门      */                               
   CREATE_DATE          DATE,                                      /*建立日期          */
   MODIFIER             VARCHAR2(12),                              /*修改人员          */
   MODI_DATE            DATE,                                      /*修改日期          */
   FLAG                 NUMBER(1),                                 /*资料状态          */
   constraint PK_TB_PMTRL primary key (PMTRL_ID)
);
create unique index AK_TB_PMTRL on TB_PMTRL (PMTRL001,PMTRL002,PMTRL003);
create sequence SEQ_TB_PMTRL minvalue 1 maxvalue 9999999999999999999999999999 start with 1 increment by 1 nocache;
grant select on TB_PMTRL to public;
grant index  on TB_PMTRL to public;
grant update on TB_PMTRL to public; 
grant delete on TB_PMTRL to public;  
grant insert on TB_PMTRL to public; 
grant select on SEQ_TB_PMTRL to public;
